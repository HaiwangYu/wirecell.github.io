84  Sequence<Val> resample(const Sequence<Val>& wave, const Domain& domain, int nsamples, const Domain& newdomain) {
85  const int oldnsamples = wave.size();
86  const double oldstep = sample_width(domain, oldnsamples);
87  const double step = sample_width(newdomain, nsamples);
88  Sequence<Val> ret;
89  for (int ind=0; ind<nsamples; ++ind) {
90  double cursor = newdomain.first + ind*step;
91  double oldfracsteps = (cursor-domain.first)/oldstep;
92  int oldind = int(oldfracsteps);
93  if (cursor <= domain.first || oldind <= 0) {
94  ret.push_back(wave[0]);
95  continue;
96  }
97  if (cursor >= domain.second or oldind+1 >= oldnsamples) {
98  ret.push_back(wave[oldnsamples-1]);
99  continue;
100  }
101  double d1 = oldfracsteps - oldstep*oldind;
102  double d2 = oldstep - d1;
103  Val newval = (wave[oldind] * d1 + wave[oldind+1]*d2) / oldstep;
104  ret.push_back(newval);
105  }
106  return ret;
107  }
