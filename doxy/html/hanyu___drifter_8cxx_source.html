110 IDepo::pointer Gen::hyDrifter::transport(IDepo::pointer depo)
111 {
112  if (!depo) {
113  cerr << "Gen::hyDrifter error, asked to transport null depo\n";
114  return nullptr;
115  }
116  // position
117  const Point here = depo->pos();
118  const Point there(m_location, here.y(), here.z());
119 
120  // time
121  const double now = depo->time();
122  const double dt = (here.x() - m_location)/m_speed;
123  const double then = now + dt;
124 
125  // number of electrons to start with
126  // ionization electrons without recombination
127  const double Qi = depo->charge();
128  // recombination probability of an electron
129  // how to initialize m_recomb? from Deposition?
130  double recombprob = m_recomb;
131  // absorption probability of an electron
132  double absorbprob = 1 - exp(-dt/m_lifetime);
133  // final number of electrons after drift
134  double dQ = Qi * recombprob * absorbprob;
135  // how many electrons remain, with fluctuation.
136  if (m_fluctuate) {
137  // fano factor
138  // use a binomial distribution with np = Qi, 1-p = m_fano
139  // this approximates a Gaussian distribution and has a physical limit >0
140  double Qi2 = random_binomial((int)Qi/(1-m_fano), 1-m_fano);
141  // binomial distribution
142  dQ = random_binomial((int)Qi2, recombprob*absorbprob);
143  }
144  const double Qf = Qi - dQ;
145 
146  const double sigmaL = sqrt(2.0*m_DL*dt);
147  const double sigmaT = sqrt(2.0*m_DT*dt);
148 
149  auto ret = make_shared<SimpleDepo>(then, there, Qf, depo, sigmaL, sigmaT);
150  // cerr << "Gen::hyDrifter: "
151  // << "q:" << ret->charge()/units::eplus << " electrons "
152  // << "t:" << depo->time()/units::us << " -> " << ret->time()/units::us << " ("<<dt/units::us<<") us, "
153  // << "x:" << depo->pos().x()/units::mm << " -> " << ret->pos().x()/units::mm << "mm, "
154  // << "long=" << ret->extent_long()/units::mm << "mm, trans=" << ret->extent_tran()/units::mm << "mm "
155  // << endl;
