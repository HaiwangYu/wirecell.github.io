125  /* This was originally Xin's cell algorithm but only the concept
126  remains.
127 
128  The idea is based on "crossing points" and "wire half-way
129  lines". Crossing points are the points in the Y-Z plane where
130  lines which are parallel to that plane cross. Wire half-way
131  lines are the lines running parallel to a given wire and within
132  the wire plane which are 1/2 the of a pitch distance away.
133 
134  The algorithm visits each U and V wire pair and determines
135  their crossing point. If this point is outside the bounding
136  box of the wire planes it is discarded. Otherwise, the four
137  crossing points of each wire half-way lines are found. Any
138  that are inside the bounding box are saved as candidate cell
139  corner points.
140 
141  Then, for each W wire, these four points are checked and any
142  which are withing 1/2 pitch of the wire are kept as cell corner
143  points.
144 
145  In addition the wire half-way crossing points for the W wire
146  and the U wire are found and checked to be within the wire
147  half-way distance with the current V wires. This is repeated
148  with U and V swapped. Surviving points are added to the list
149  of cell corner points.
150 
151  */
152 
153  IWire::vector u_wires, v_wires, w_wires;
154  copy_if(wires->begin(), wires->end(), back_inserter(u_wires), select_u_wires);
155  copy_if(wires->begin(), wires->end(), back_inserter(v_wires), select_v_wires);
156  copy_if(wires->begin(), wires->end(), back_inserter(w_wires), select_w_wires);
157 
158  std::sort(u_wires.begin(), u_wires.end(), ascending_index);
159  std::sort(v_wires.begin(), v_wires.end(), ascending_index);
160  std::sort(w_wires.begin(), w_wires.end(), ascending_index);

163  const Ray pitch_v_ray = pitch2(v_wires);
164  const Ray pitch_w_ray = pitch2(w_wires);

166  const double pitch_u = ray_length(pitch_u_ray);
167  const double pitch_v = ray_length(pitch_v_ray);
168  const double pitch_w = ray_length(pitch_w_ray);

170  const Vector axis_u = axis(u_wires);
171  const Vector axis_v = axis(v_wires);
172  const Vector axis_w = axis(w_wires);
173 
174  const Vector jump_u = axis_u * (pitch_u / sin(2.0 * acos(axis_w.dot(axis_u))));
175  const Vector jump_v = axis_v * (pitch_v / sin(2.0 * acos(axis_w.dot(axis_v))));
176 
177  // jumps from a wire crossing point to the four half-wire crossing
178  // points, in cyclical order.
179  const Vector cross_jump_uv[4] = {
180  + 0.5*jump_u + 0.5*jump_v, // +Y/Z=0
181  + 0.5*jump_u - 0.5*jump_v, // Y=0/+Z
182  - 0.5*jump_u - 0.5*jump_v, // -Y/Z=0
183  - 0.5*jump_u + 0.5*jump_v // Y=0/-Z
184  };

190

192  for (auto wire : *wires) {
193  bb(wire->ray());
194  }
195 
196  // pack it up and send it out
197  std::vector<IWire::pointer> uvw_wires(3);
198 
199  const Vector origin_uv = origin_cross(u_wires, v_wires);

202  //const double last_w_z = w_wires[w_wires.size()-1]->ray().first.z();
203  const double w_lane_half_width = 0.5*(pitch_w + tolerance);
204 
205  for (size_t u_ind = 0; u_ind < u_wires.size(); ++u_ind) {
206  IWire::pointer u_wire = u_wires[u_ind];
207  uvw_wires[0] = u_wire;

210  IWire::pointer v_wire = v_wires[v_ind];
211  uvw_wires[1] = v_wire;
212

214  Vector cross_uv = origin_uv - double(u_ind)*jump_v + double(v_ind)*jump_u;
215 
216  if (cross_uv.z() < bb.bounds().first.z() - 0.5*pitch_w ||
217  cross_uv.z() > bb.bounds().second.z() + 0.5*pitch_w ||
218  cross_uv.y() < bb.bounds().first.y() - y_buffer ||
219  cross_uv.y() > bb.bounds().second.y() + y_buffer)
220  {
221  continue;
222  }

233

239

246  inside_uv[cjind] = is_inlane;

250  }
251

253  continue;
254  }
255

257  std::vector<std::pair<int,int> > to_break;
258  for (int cjind=0; cjind<4; ++cjind) {
259  int other_ind = (cjind+1)%4;
260  if (inside_uv[cjind] && inside_uv[other_ind]) {
261  continue; // both inside
262  }
263  if (!(inside_uv[cjind] || inside_uv[other_ind])) {
264  continue; // both outside
265  }
266  if (inside_uv[cjind]) { // order inside-first
267  to_break.push_back(std::pair<int,int>(cjind, other_ind));
268  }

271  }
272  }
273 
274  for (size_t tbind = 0; tbind < to_break.size(); ++tbind) {
275  const Vector& p_in = puv[to_break[tbind].first];
276  const Vector& p_out = puv[to_break[tbind].second];
277 
278  double break_z = 0.0;
279  if (p_out.z() > target_w_z) {
280  break_z = target_w_z + 0.5*pitch_w - p_in.z();
281  }
282  else {
283  break_z = target_w_z - 0.5*pitch_w - p_in.z();
284  }
285 
286  Vector diff = p_out - p_in;
287  double rel = break_z / diff.z();
288  Vector corner = p_in + rel*diff;
289  pcell.push_back(corner);
290  }
291 
292  // result
293  res_cells->push_back(ICell::pointer(new BoundCell(res_cells->size(), pcell, uvw_wires)));
294 
295  } // W wires
296  } // v wires
297  } // u wires

300  return true;
