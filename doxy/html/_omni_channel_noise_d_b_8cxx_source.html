95 std::vector<int> OmniChannelNoiseDB::parse_channels(const Json::Value& jchannels)
96 {
97  std::vector<int> ret;
98 
99  // single channel
100  if (jchannels.isInt()) {
101  ret.push_back(jchannels.asInt());
102  return ret;
103  }
104 
105  // array of explicit channels
106  if (jchannels.isArray()) {
107  const int nch = jchannels.size();
108  ret.resize(nch);
109  for (int ind=0; ind<nch; ++ind) {
110  ret[ind] = jchannels[ind].asInt();
111  }
112  return ret;
113  }
114 
115  // else, assume an object
116 
117  // range
118  if (jchannels.isMember("first") && jchannels.isMember("last")) {
119  const int chf = jchannels["first"].asInt();
120  const int chl = jchannels["last"].asInt();
121  const int nch = chl-chf+1;
122  ret.resize(nch);
123  for (int ind=0; ind < nch; ++ind) {
124  ret[ind] = chf + ind;
125  }
126  return ret;
127  }
128 
129  // wire plane id
130  if (jchannels.isMember("wpid")) {
131  WirePlaneId wpid(jchannels["wpid"].asInt());
132  for (auto ch : m_anode->channels()) {
133  if (m_anode->resolve(ch) == wpid) {
134  ret.push_back(ch);
135  }
136  }
137  return ret;
138  }
139 
140  return ret;

312 OmniChannelNoiseDB::shared_filter_t OmniChannelNoiseDB::parse_response(Json::Value jfilt)
313 {
314  if (jfilt.isMember("wpid")) {
315  WirePlaneId wpid(jfilt["wpid"].asInt());
316  auto it = m_response_cache.find(wpid.ident());
317  if (it != m_response_cache.end()) {
318  return it->second;
319  }
320  auto const& fr = m_fr->field_response();
321  auto fravg = Response::wire_region_average(fr);
322  auto const& pr = fravg.planes[wpid.index()];
323 
324  // full length waveform
325  std::vector<float> waveform(m_nsamples, 0.0);
326  for (auto const& path : pr.paths) {
327  auto const& current = path.current;
328  const size_t nsamp = std::min(m_nsamples, (int)current.size());
329  for (size_t ind=0; ind<nsamp; ++ind) {
330  waveform[ind] += current[ind];
331  }
332  }
333  auto spectrum = WireCell::Waveform::dft(waveform);
334  auto ret = std::make_shared<filter_t>(spectrum);
335  m_response_cache[wpid.ident()] = ret;
336  return ret;
337  }
338 
339  if (jfilt.isMember("waveform") && jfilt.isMember("waveformid")) {
340  int id = jfilt["waveformid"].asInt();
341  auto it = m_waveform_cache.find(id);
342  if (it != m_waveform_cache.end()) {
343  return it->second;
344  }
345 
346  auto jwave = jfilt["waveform"];
347  const int nsamp = std::min(m_nsamples, (int)jwave.size());
348 
349  // Explicitly given waveform
350  std::vector<float> waveform(m_nsamples, 0.0);
351  for (int ind=0; ind<nsamp; ++ind) {
352  waveform[ind] = jwave[ind].asFloat();
353  }
354 
355  auto spectrum = WireCell::Waveform::dft(waveform);
356  auto ret = std::make_shared<filter_t>(spectrum);
357  m_waveform_cache[id] = ret;
358  return ret;
359  }
360 
361  // this default return is special in that it's empty instead of
362  // being a flat, unity spectrum.
363  static shared_filter_t empty = std::make_shared<filter_t>();
364  return empty;
